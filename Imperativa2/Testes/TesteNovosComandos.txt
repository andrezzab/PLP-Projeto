LOAD "Testes/csvs/teste_drop.csv" AS dados;
MEDIAN dados.idade AS mediana_idade;
MEDIAN dados.salario AS mediana_salario;
MODE dados.idade AS moda_idade;
MODE dados.salario AS moda_salario;
STD dados.idade AS desvio_idade;
STD dados.salario AS desvio_salario;
VARIANCE dados.idade AS variancia_idade;
VARIANCE dados.salario AS variancia_salario;
MIN dados.idade AS minimo_idade;
MIN dados.salario AS minimo_salario;
MAX dados.idade AS maximo_idade;
MAX dados.salario AS maximo_salario
