// Primeiro, carregue o CSV para a variável 'dados'
LOAD "Testes/csvs/teste_drop.csv" AS dados;

// Em seguida, execute as estatísticas na coluna 'idade' da variável 'dados'
STATS dados ON "idade";

// Você também pode calcular para outra coluna
STATS dados ON "salario";

// 1. Conta quantos funcionários existem no total
COUNT dados;

// 2. Filtra apenas os funcionários de "Recife" e salva em uma nova variável
FILTER dados INTO funcionarios_recife WHERE cidade == "recife"
